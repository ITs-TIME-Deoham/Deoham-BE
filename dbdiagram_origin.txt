Table users {
  id bigint [pk, increment]
  email varchar(255) [unique, note: '가입 시 provider_email에서 복사. Apple "이메일 가리기" 선택 시 null 가능']
  phone varchar(20) [unique, note: '소셜 로그인으로는 수집 불가. 사용자가 프로필에서 직접 입력']
  nickname varchar(50) [not null]
  age int [not null]
  profile_image_url varchar(500)
  is_verified boolean [not null, default: false]
  avg_rating float [not null, default: 0, note: '현재 reviews 테이블 없음 — 갱신 시점 및 방식 추후 결정 필요']
  matching_count int [not null, default: 0]
  role user_role [not null, default: 'user']
  status user_status [not null, default: 'active']
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]

  Note: '회원 마스터. 인증은 user_social_accounts에서 관리'
}

Table user_social_accounts {
  id bigint [pk, increment]
  user_id bigint [not null, ref: > users.id]
  provider oauth_provider [not null]
  provider_user_id varchar(255) [not null, note: 'Apple: sub, Kakao: id (숫자형 문자열)']
  provider_email varchar(255) [note: '프로바이더 제공 이메일. Apple은 최초 로그인 시에만 전달됨']
  access_token text [note: '암호화 저장 권장 (AES-256 등)']
  refresh_token text [note: '암호화 저장 권장']
  token_expires_at timestamp
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]

  Indexes {
    (provider, provider_user_id) [unique, note: '로그인 시 사용자 조회 핵심 인덱스']
    user_id
  }

  Note: '소셜 로그인 계정. 한 회원이 Apple + Kakao 동시 연결 가능. 신규 프로바이더 추가 시 oauth_provider enum만 확장'
}

Table user_locations {
  user_id bigint [pk, ref: - users.id]
  lat decimal(10,7) [not null]
  lng decimal(10,7) [not null]
  city varchar(100)
  updated_at timestamp [not null, default: `now()`]

  Indexes {
    (lat, lng) [note: '운영시 PostGIS GIST 인덱스 권장']
  }

  Note: '실시간 위치. 반경 푸시용. users 와 1:1 로 분리해 쓰기 부하 격리'
}

Table sos_requests {
  id bigint [pk, increment]
  requester_id bigint [not null, ref: > users.id]
  category sos_category [not null]
  title varchar(100) [not null]
  description text
  location_lat decimal(10,7) [not null]
  location_lng decimal(10,7) [not null]
  location_name varchar(200)
  radius_m int [not null, default: 500]
  max_participants int [not null, default: 1]
  status sos_status [not null, default: 'open']
  expires_at timestamp [not null]
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]

  Indexes {
    (status, expires_at) [note: '활성 요청 조회']
    requester_id
    (location_lat, location_lng) [note: '운영시 PostGIS GIST 권장']
  }
}

Table request_participations {
  id bigint [pk, increment]
  request_id bigint [not null, ref: > sos_requests.id]
  user_id bigint [not null, ref: > users.id]
  status participation_status [not null, default: 'pending']
  applied_at timestamp [not null, default: `now()`]
  responded_at timestamp

  Indexes {
    (request_id, user_id) [unique, note: '한 요청에 한 사용자 중복 신청 방지']
    user_id
  }
}

Table chat_rooms {
  id bigint [pk, increment]
  request_id bigint [unique, not null, ref: > sos_requests.id]
  status chat_room_status [not null, default: 'active']
  created_at timestamp [not null, default: `now()`]
  closed_at timestamp
}

Table chat_messages {
  id bigint [pk, increment]
  chat_room_id bigint [not null, ref: > chat_rooms.id]
  sender_id bigint [not null, ref: > users.id]
  content text [not null]
  message_type chat_message_type [not null, default: 'text']
  sent_at timestamp [not null, default: `now()`]
  read_at timestamp

  Indexes {
    (chat_room_id, sent_at) [note: '방별 시간순 조회']
  }
}

Table reports {
  id bigint [pk, increment]
  reporter_id bigint [not null, ref: > users.id]
  reported_user_id bigint [not null, ref: > users.id]
  request_id bigint [ref: > sos_requests.id]
  reason_type report_reason [not null]
  description text
  status report_status [not null, default: 'pending']
  created_at timestamp [not null, default: `now()`]
  resolved_at timestamp

  Indexes {
    reported_user_id
    status
  }
}

Table push_logs {
  id bigint [pk, increment]
  request_id bigint [not null, ref: > sos_requests.id]
  user_id bigint [not null, ref: > users.id]
  sent_at timestamp [not null, default: `now()`]

  Indexes {
    (request_id, user_id) [unique, note: '같은 SOS에 중복 발송 방지']
  }

  Note: '푸시 알림 발송 기록. 중복 발송 방지용'
}

Table notices {
  id bigint [pk, increment]
  author_id bigint [not null, ref: > users.id, note: '공지를 작성한 운영자 (users 테이블 참조)']
  type notice_type [not null, default: 'general']
  title varchar(200) [not null]
  content text [not null]
  is_published boolean [not null, default: false, note: '임시저장(false) → 게시(true) 전환용']
  published_at timestamp [note: 'is_published가 true로 바뀐 시각. 게시 전에는 null']
  created_at timestamp [not null, default: `now()`]
  updated_at timestamp [not null, default: `now()`]

  Indexes {
    author_id
    (is_published, published_at) [note: '게시된 공지 최신순 조회']
  }

  Note: '공지사항. author_id로 작성자(운영자) 추적. is_published=false는 임시저장 상태'
}

Table notice_reads {
  id bigint [pk, increment]
  notice_id bigint [not null, ref: > notices.id]
  user_id bigint [not null, ref: > users.id]
  read_at timestamp [not null, default: `now()`]

  Indexes {
    (notice_id, user_id) [unique, note: '한 사용자가 같은 공지를 중복 읽음 처리 방지']
    user_id [note: '사용자가 읽은 공지 목록 조회']
  }

  Note: '공지 읽음 여부 추적. 읽지 않은 공지 수 뱃지 등에 활용'
}
